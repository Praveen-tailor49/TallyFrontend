      function numberToWords(num) {
        const ones = [
          "",
          "One",
          "Two",
          "Three",
          "Four",
          "Five",
          "Six",
          "Seven",
          "Eight",
          "Nine",
          "Ten",
          "Eleven",
          "Twelve",
          "Thirteen",
          "Fourteen",
          "Fifteen",
          "Sixteen",
          "Seventeen",
          "Eighteen",
          "Nineteen",
        ];
        const tens = [
          "",
          "",
          "Twenty",
          "Thirty",
          "Forty",
          "Fifty",
          "Sixty",
          "Seventy",
          "Eighty",
          "Ninety",
        ];

        if (num === 0) return "Zero";

        function convertHundreds(n) {
          if (n < 20) return ones[n];
          if (n < 100)
            return (
              tens[Math.floor(n / 10)] + (n % 10 ? " " + ones[n % 10] : "")
            );
          return (
            ones[Math.floor(n / 100)] +
            " Hundred" +
            (n % 100 ? " " + convertHundreds(n % 100) : "")
          );
        }

        function convert(n) {
          if (n < 1000) return convertHundreds(n);
          if (n < 100000)
            return (
              convertHundreds(Math.floor(n / 1000)) +
              " Thousand" +
              (n % 1000 ? " " + convertHundreds(n % 1000) : "")
            );
          if (n < 10000000)
            return (
              convertHundreds(Math.floor(n / 100000)) +
              " Lakh" +
              (n % 100000 ? " " + convert(n % 100000) : "")
            );
          return (
            convertHundreds(Math.floor(n / 10000000)) +
            " Crore" +
            (n % 10000000 ? " " + convert(n % 10000000) : "")
          );
        }

        return (
          "INR " +
          convert(Math.floor(num)) +
          (num % 1 ? " and " + Math.round((num % 1) * 100) + " Paise" : "") +
          " Only"
        );
      }

      function formatDate(dateStr) {
        if (!dateStr) return "";
        const date = new Date(dateStr);
        const day = date.getDate();
        const months = [
          "Jan",
          "Feb",
          "Mar",
          "Apr",
          "May",
          "Jun",
          "Jul",
          "Aug",
          "Sep",
          "Oct",
          "Nov",
          "Dec",
        ];
        const month = months[date.getMonth()];
        const year = date.getFullYear();
        return `${day}-${month}-${year}`;
      }

      function downloadPDF() {
        try {
          // Check html2pdf loaded
          if (typeof html2pdf === "undefined") {
            alert("PDF library (html2pdf) load nahi hui. Path check karo.");
            return;
          }

          const element = document.getElementById("invoicePage");
          if (!element) {
            alert("Invoice preview element nahi mila. Pehle 'Generate Invoice' karo.");
            return;
          }

          // Send download count message to React Native
          if (window.ReactNativeWebView) {
            window.ReactNativeWebView.postMessage(
              JSON.stringify({ type: "incrementDownloadCount" }),
            );
          }

          const opt = {
            margin: 0,
            filename: `Invoice_${document.getElementById("invoiceNo").value}.pdf`,
            image: { type: "jpeg", quality: 0.98 },
            html2canvas: {
              scale: 2,
              useCORS: true,
              logging: false,
              letterRendering: true,
              allowTaint: true,
              windowWidth: element.offsetWidth,
              windowHeight: element.offsetHeight,
            },
            jsPDF: {
              unit: "mm",
              format: "a4",
              orientation: "portrait",
              compress: true,
            },
            pagebreak: { mode: "avoid-all" },
          };

          html2pdf()
            .set(opt)
            .from(element)
            .toPdf()
            .get("pdf")
            .then(function (pdf) {
              try {
                const pdfData = pdf.output("datauristring");
                if (window.ReactNativeWebView) {
                  window.ReactNativeWebView.postMessage(
                    JSON.stringify({
                      type: "sharePDF",
                      data: pdfData,
                      filename: `Invoice_${document.getElementById("invoiceNo").value}.pdf`,
                    }),
                  );
                } else {
                  pdf.save();
                }
              } catch (innerErr) {
                console.error("PDF save error:", innerErr);
                alert("PDF save error: " + (innerErr && innerErr.message ? innerErr.message : innerErr));
              }
            })
            .catch(function (err) {
              console.error("html2pdf error:", err);
              alert("PDF generate error: " + (err && err.message ? err.message : err));
            });
        } catch (err) {
          console.error("downloadPDF error:", err);
          alert("Download PDF error: " + (err && err.message ? err.message : err));
        }
      }

      // Initialize - ensure form is shown on page load
      document.getElementById("formContainer").style.display = "block";
      document.getElementById("previewContainer").classList.remove("active");
      renderItemsForm();

      // Load Sellers Dropdown
      async function loadSellersDropdown() {
        try {
          const response = await fetch('https://tallybackend-rjib.onrender.com/api/sellers');
          const data = await response.json();
          const sellerDropdown = document.getElementById('sellerDropdown');

          // Clear existing options
          sellerDropdown.innerHTML = '';

          data.sellers.forEach(seller => {
            const option = document.createElement('option');
            option.value = seller.id;
            option.textContent = seller.companyName;
            sellerDropdown.appendChild(option);
          });

          // Select first option if available
          if (data.sellers.length > 0) {
            sellerDropdown.selectedIndex = 0;
            loadSellerData();
          }
        } catch (error) {
          console.error('Error loading sellers:', error);
        }
      }

      // Save Seller Data
      async function saveSellerData() {
        const companyName = document.getElementById('sellerCompany').value;
        
        // Check if seller with same company name already exists
        const sellerDropdown = document.getElementById('sellerDropdown');
        let companyExists = false;
        for (let i = 0; i < sellerDropdown.options.length; i++) {
          if (sellerDropdown.options[i].textContent.toLowerCase() === companyName.toLowerCase()) {
            companyExists = true;
            break;
          }
        }
        
        if (companyExists) {
          alert('Seller with this company name already exists!');
          return;
        }
        
        const sellerData = {
          id: Date.now().toString(),
          name: companyName,
          companyName: companyName,
          address: document.getElementById('sellerAddress').value,
          gstin: document.getElementById('sellerGSTIN').value,
          stateName: document.getElementById('sellerState').value,
          stateCode: document.getElementById('sellerStateCode').value,
          contact: document.getElementById('sellerContact').value,
          email: document.getElementById('sellerEmail').value,
        };

        if (window.ReactNativeWebView) {
          window.ReactNativeWebView.postMessage(
            JSON.stringify({
              type: 'saveSellerData',
              sellerData: sellerData,
            }),
          );
        } else {
          // Fallback for web platform - make direct API call
          try {
            const token = localStorage.getItem('token');
            // Decode JWT token to get userId
            const tokenPayload = JSON.parse(atob(token.split('.')[1]));
            const userId = tokenPayload.userId;
            
            const response = await fetch('https://tallybackend-rjib.onrender.com/api/seller', {
              method: 'POST',
              headers: {
                'Content-Type': 'application/json',
                'Authorization': `Bearer ${token}`,
              },
              body: JSON.stringify({
                ...sellerData,
                userId,
              }),
            });
            
            if (response.ok) {
              alert('Seller data saved successfully!');
              loadSellersDropdown();
            } else {
              const errorData = await response.json();
              alert('Failed to save seller data: ' + (errorData.error || 'Unknown error'));
            }
          } catch (error) {
            console.error('Error saving seller data:', error);
            alert('Error saving seller data: ' + error.message);
          }
        }
      }
